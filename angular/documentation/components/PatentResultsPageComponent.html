import { Component, OnInit } from '@angular/core';

@Component({
  selector: 'app-patent-results-page',
  templateUrl: './patent-results-page.component.html',
  styleUrls: ['./patent-results-page.component.scss']
})
export class PatentResultsPageComponent implements OnInit {

  constructor() { }

  private loadScripts() {
    let inputScript = document.createElement("script");
    inputScript.type = "text/javascript";
    inputScript.src = "../assets/scripts/input-script.js";
    document.body.appendChild(inputScript);
  }

  ngOnInit(): void {
    this.loadScripts();
  }

}
