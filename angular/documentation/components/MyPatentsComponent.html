import {Component, OnInit} from '@angular/core';

@Component({
  selector: 'app-my-patents',
  templateUrl: './my-patents.component.html',
  styleUrls: ['./my-patents.component.scss']
})
export class MyPatentsComponent implements OnInit {

  constructor() {
  }

  private loadScripts() {
    let inputScript = document.createElement("script");
    inputScript.type = "text/javascript";
    inputScript.src = "../../assets/scripts/input-script.js";
    document.body.appendChild(inputScript);
  }

  ngOnInit(): void {
    this.loadScripts();
  }
}
